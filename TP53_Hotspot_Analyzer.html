# 2. Sequence Retrieval via Biopython

def fetch_seq_fasta(ncbi_id = "NM_000546.6"):
    """
    Fetch the TP53 coding sequence (CDS) from NCBI using Biopython Entrez.
    Parameters
    ----------
    ncbi_id : str
        RefSeq accession ID for TP53 transcript (default NM_000546.6).
    Returns
    -------
    record : Bio.SeqRecord.SeqRecord
        The nucleotide sequence in FASTA format as a SeqRecord object.
    """
    try:
        # Query the NCBI nucleotide database and request the sequence in FASTA format.
        handle = Entrez.efetch(db = "nucleotide", id = ncbi_id, rettype = "fasta", retmode = "text")
        
        # Parse the returned FASTA text into a Biopython SeqRecord for structured access.
        record = SeqIO.read(StringIO(handle.read()), "fasta")
        handle.close()
        
        # Confirm successful retrieval and print basic metadata.
        print("Successfully fetched {} ({} nt)".format(ncbi_id, len(record.seq)))
        return record
    
    except Exception as e:
        # Catch any connection or parsing errors and present a clear runtime message.
        raise RuntimeError("Unable to fetch TP53 sequence: {}".format(e))

# Retrieve the TP53 nucleotide coding sequence.
tp53_record = fetch_seq_fasta()

# Extract the raw nucleotide sequence from the SeqRecord.
seq = tp53_record.seq

# Trim the sequence so its length is a multiple of three.
# This ensures complete codons for translation and avoids partial triplets at the end.
seq = seq[:len(seq) - (len(seq) % 3)]

# Translate the trimmed CDS to obtain the TP53 protein sequence.
# The `to_stop=True` flag halts translation at the first stop codon.
tp53_protein = seq.translate(to_stop=True)

# Display key properties of the sequence for reference.
print("CDS length: {}".format(len(seq)))              # number of nucleotides
print("Protein length: {}".format(len(tp53_protein))) # number of amino acids
print("{} ...".format(str(tp53_protein)[:60]))        # preview of translated sequence

# ============================================================
# 3. ClinVar Variant Retrieval
# Goal: Load TP53 variants from ClinVars public summary table (tab-delimited, gzipped),
#       extract protein-level HGVS notations ('p.'), and retain interpretable condition labels.
# Rationale: The ClinVar schema evolves; column names can vary. We detect columns programmatically
#            to keep the notebook resilient to upstream format changes.
# ============================================================

def load_clinvar_tp53_from_summary(url=None, assembly="GRCh38"):
    """Return a tidy TP53 variant table: Protein_Change, Cancer_Type, Source."""
    # Default to the official NCBI ClinVar summary (compressed TSV).
    if url is None:
        url = "https://ftp.ncbi.nlm.nih.gov/pub/clinvar/tab_delimited/variant_summary.txt.gz"

    # --- Header sniff: read only the header to discover exact column names (no data yet). ---
    header = pd.read_csv(
        url, sep="\t", compression="infer", nrows=0, dtype=str
    )
    cols = list(header.columns)  # explicit list for predictable iteration

    # --- Helper: find the first column whose name matches any provided pattern (case-insensitive). ---
    def find_col(cols, patterns):
        for p in patterns:
            for c in cols:
                if re.search(p, c, flags=re.IGNORECASE):  # tolerate schema drift
                    return c
        return None  # not found  caller handles gracefully

    # --- Dynamically resolve relevant columns; names vary across ClinVar releases. ---
    gene_col    = find_col(cols, ["GeneSymbol"])
    assembly_col= find_col(cols, ["Assembly"])
    name_col    = find_col(cols, ["Name"])
    prot_col    = find_col(cols, ["Protein", "HGVS_p"])              # protein-level HGVS if present
    clin_col    = find_col(cols, ["ClinicalSignificance"])           # may contain disease words
    pheno_col   = find_col(cols, ["PhenotypeList","Condition","Trait","PhenotypeIDS"])  # preferred for readable labels

    # --- Read only what we need to minimize I/O; usecols avoids loading the entire file. ---
    usecols = [c for c in [gene_col, assembly_col, name_col, prot_col, clin_col, pheno_col] if c]
    df_all = pd.read_csv(
        url, sep="\t", compression="infer", usecols=usecols, dtype=str, low_memory=False
    )

    # --- Biological subsetting: keep only TP53 entries; optionally match the target assembly. ---
    df = df_all[df_all[gene_col] == "TP53"].copy()
    if assembly_col in df.columns:
        df = df[df[assembly_col] == assembly].copy()

    # --- Protein HGVS extraction: prefer explicit protein column; otherwise parse from 'Name'. ---
    if prot_col in df.columns:
        df["Protein_Change"] = df[prot_col]
    else:
        # Extract the first 'p.' token anywhere in 'Name'; ensures we retain true protein-level notations.
        df["Protein_Change"] = df[name_col].astype(str).str.extract(r"(p\.[A-Za-z0-9*]+)", expand=False)

    # --- Keep only rows that actually have protein-level HGVS notations. ---
    df = df[df["Protein_Change"].astype(str).str.startswith("p.")].copy()

    # --- Phenotype/condition text clean-up for interpretability in tables/plots. ---
    if pheno_col in df.columns:
        # 1) Convert to string to avoid NA-propagation issues in regex operations.
        cleaned = df[pheno_col].astype(str)
        # 2) Remove database identifier prefixes (e.g., "MedGen:1234|OMIM:151623") to expose human-readable labels.
        cleaned = cleaned.str.replace(
            r"(Gene|MedGen|OMIM|Orphanet|MONDO|HP|MeSH|DOID)[:0-9,]+", "", regex=True
        )
        # 3) Normalize separators (vertical bars or repeated delimiters) and trim stray punctuation/whitespace.
        cleaned = cleaned.str.replace(r"\|+", ", ", regex=True).str.strip(",; |")
        df["Cancer_Type"] = cleaned
    elif clin_col in df.columns:
        # Fallback: use clinical significance text when dedicated phenotype fields are absent/empty.
        df["Cancer_Type"] = df[clin_col]
    else:
        # Last resort: mark empty; will be removed below.
        df["Cancer_Type"] = ""

    # --- Remove non-informative labels to keep downstream summaries biologically meaningful. ---
    df["Cancer_Type"] = df["Cancer_Type"].fillna("").astype(str)
    df = df[df["Cancer_Type"].str.strip().ne("")]                               # drop blank
    df = df[~df["Cancer_Type"].str.contains("not provided", case=False)]        # drop not provided
    df = df[~df["Cancer_Type"].str.contains("NA", case=False)]                  # drop NA tokens

    # --- Provenance tag and de-duplication to avoid double counting. ---
    df["Source"] = "ClinVar"
    df = df[["Protein_Change", "Cancer_Type", "Source"]].drop_duplicates().reset_index(drop=True)

    # Diagnostic print for reproducibility and quick sanity check in notebooks.
    print("Loaded {} TP53 variants with {} distinct cancer labels".format(len(df), df["Cancer_Type"].nunique()))
    return df

# ============================================================
# 4. Variant Parsing (HGVS 'p.' notations)
# Goal: Convert free-text protein changes into structured fields for analysis.
#       Consequence categories: missense, nonsense, synonymous, frameshift, other.
# Notes:
#    Handles both one-letter (p.R175H) and three-letter (p.Arg175His) forms.
#    Accepts optional parentheses used in some ClinVar records.
#    Frameshifts (e.g., p.Asn210fs) are captured explicitly and labeled.
# ============================================================

# Map HGVS three-letter residue names to one-letter codes for consistency in tables/plots.
AA_THREE_TO_ONE = {
    'Ala':'A','Arg':'R','Asn':'N','Asp':'D','Cys':'C','Gln':'Q','Glu':'E','Gly':'G',
    'His':'H','Ile':'I','Leu':'L','Lys':'K','Met':'M','Phe':'F','Pro':'P','Ser':'S',
    'Thr':'T','Trp':'W','Tyr':'Y','Val':'V','Ter':'*','Stop':'*'
}

# Substitution regex patterns:
#   RE_ONE   : one-letter form (e.g., p.R175H or p.*123A)
#   RE_THREE : three-letter form (e.g., p.Arg175His or p.Arg213Ter)
RE_ONE        = re.compile(r'^\(?p\.([A-Z*])(\d+)([A-Z*])\)?$')
RE_THREE      = re.compile(r'^\(?p\.([A-Z][a-z]{2})(\d+)([A-Z][a-z]{2}|\*)\)?$')

# Frameshift regex patterns:
#   RE_FS_ONE   : one-letter frameshift (p.N210fs)
#   RE_FS_THREE : three-letter frameshift (p.Asn210fs)
RE_FS_ONE     = re.compile(r'^\(?p\.([A-Z*])(\d+)fs', re.IGNORECASE)
RE_FS_THREE   = re.compile(r'^\(?p\.([A-Z][a-z]{2})(\d+)fs', re.IGNORECASE)

def parse_hgvs_p(p):
    """
    Parse an HGVS protein string.
    Returns a dict with AA_Pos (int or None), Ref_AA, Alt_AA (None for frameshift),
    and Consequence in {'missense','nonsense','synonymous','frameshift','other'}.
    """
    # EAFP guard: tolerate non-string inputs without halting the workflow.
    try:
        p = p.strip()
    except Exception:
        return {"AA_Pos": None, "Ref_AA": None, "Alt_AA": None, "Consequence": None}

    # Explicit synonymy indicator occasionally present in HGVS.
    if p == "p.=":
        return {"AA_Pos": None, "Ref_AA": None, "Alt_AA": None, "Consequence": "synonymous"}

    # --- Frameshift detection is prioritized to avoid misclassification as substitution. ---
    m = RE_FS_ONE.match(p)
    if m:
        ref1, pos = m.groups()
        return {"AA_Pos": int(pos), "Ref_AA": ref1, "Alt_AA": None, "Consequence": "frameshift"}
    
    m = RE_FS_THREE.match(p)
    if m:
        ref3, pos = m.groups()
        ref1 = AA_THREE_TO_ONE.get(ref3)
        return {"AA_Pos": int(pos), "Ref_AA": ref1, "Alt_AA": None, "Consequence": "frameshift"}

    # --- Substitution parsing (one-letter then three-letter) ---
    m = RE_ONE.match(p)
    if m:
        ref, pos, alt = m.groups()
        pos = int(pos)
        # Nonsense if '*' appears in the alternate residue;
        # synonymous if ref == alt; otherwise missense.
        cons = "nonsense" if alt == "*" else ("synonymous" if ref == alt else "missense")
        return {"AA_Pos": pos, "Ref_AA": ref, "Alt_AA": alt, "Consequence": cons}

    m = RE_THREE.match(p)
    if m:
        ref3, pos, alt3 = m.groups()
        pos = int(pos)
        ref = AA_THREE_TO_ONE.get(ref3)
        alt = "*" if alt3 in ("*", "Ter", "Stop") else AA_THREE_TO_ONE.get(alt3)
        if ref and alt:
            cons = "nonsense" if alt == "*" else ("synonymous" if ref == alt else "missense")
            return {"AA_Pos": pos, "Ref_AA": ref, "Alt_AA": alt, "Consequence": cons}

    # Complex events (e.g., delins/dup) fall here; we keep them as 'other' and do not assign a position.
    return {"AA_Pos": None, "Ref_AA": None, "Alt_AA": None, "Consequence": "other"}

# --- Vectorized parsing: build a list of dicts then DataFrame. ---

# ============================================================
# 5. Mutation Hotspot Visualization
# Goal: Identify positions with recurrent variation in TP53, including truncating frameshifts.
# Strategy:
#    Exclude rows without a resolvable AA_Pos (unparseable/complex HGVS).
#    Restrict to three consequence classes relevant for hotspot analysis:
#       missense (amino-acid change), nonsense (premature stop), frameshift (coding shift).
#    Aggregate counts per position and render a stem (lollipop) plot.
# ============================================================

# Inline filter: valid positions only + selected consequence classes.
df_plot = df_parsed.dropna(subset=["AA_Pos"]).copy()                       # remove events with no AA position
df_plot["AA_Pos"] = df_plot["AA_Pos"].astype(int)                          # ensure integer for plotting/grouping
df_plot = df_plot[df_plot["Consequence"].isin(["missense","nonsense","frameshift"])]

# Group by position and consequence; fill missing classes with zeros for consistent columns.
summary = (
    df_plot.groupby(["AA_Pos", "Consequence"])
           .size()
           .unstack(fill_value=0)
           .reset_index()
)

# Total events per codon (across the three categories) for ranking/labels.
summary["Total_Count"] = summary.drop(columns="AA_Pos").sum(axis=1)
summary = summary.sort_values("Total_Count", ascending=False)

# --- Stem (lollipop) plot to emphasize recurrence at specific positions. ---
plt.figure(figsize=(14, 4))

# Vertical stems
for pos, count in zip(summary["AA_Pos"], summary["Total_Count"]):
    plt.plot([pos, pos], [0, count], color="steelblue", lw=1)

# Marker heads
for pos, count in zip(summary["AA_Pos"], summary["Total_Count"]):
    plt.plot(pos, count, "o", color="darkred", markersize=4)

plt.title("TP53 Mutation Hotspots (ClinVar Data)")
plt.xlabel("Amino Acid Position")
plt.ylabel("Mutation Count")

# Annotate canonical hotspots for interpretability in course reports.
for hot in [175, 248, 273]:

        y_val = summary.loc[summary["AA_Pos"] == hot, "Total_Count"].values[0]
